93 def post(self, request, **kwargs): # handles POST requests 

94 genre = request.POST['genre'] # genre selected by user in the creation form 

95 year = request.POST['year'] # year selected by user in the creation form 

96 

97 # Discogs API for Python, for more information access: 

98 # https://www.discogs.com/developers and https://github.com/joalla/discogs_client 

99 import discogs_client 

100 

101 # authenticated queries to Discogs API require a personal user token. 

102 # here, the token is retrieved from a git-ignored file at the root of the project. 

103 # even without authentication, it is still possible to query their database, 

104 # but some pieces of information may be missing from the results. 

105 d = discogs_client.Client('dorsetMusicCollection/0.1', 

106 user_token=os.getenv('DISCOGS_USER_TOKEN')) 

107 

108 # query Discogs' database with provided genre and year 

109 results = d.search(type='master', style=genre, year=year) 

110 

111 if len(results) == 0: # if query returned empty 

112 # render no_matches template 

113 return render(request, 'polls/no_matches.html', context={'genre': genre, 'year': year}) 

114 else: # if query is not empty 

115 self.model = Question.objects.create() # create an entry in the database for this new poll 

116 self.model.genre = genre # set its genre 

117 self.model.year = year # set its year 

118 self.model.text = f'What is the best {genre} album of {year}?' # set its text 

119 self.model.save() # save it to database 

120 

121 # combines all results' pages into one list 

122 all_pages = [(results.page(i)) for i in range(results.pages)] 

123 # only master release records from each page in new list 

124 all_pages_flattened = [master for page in all_pages for master in page] 

125 

126 masters = {} 

127 for master in all_pages_flattened: 

128 

129 artist = master.title.split(' - ')[0] # set artist from master 

130 artists = [v[4] for k, v in masters.items()] # list of artists being handled 

131 

132 if artist in artists: # checks if the current artist is already in the list 

133 artist_index = artists.index(artist) 

134 

135 # if current artist is already in the list, keep only their most popular release 

136 if master.data['community']['have'] + master.data['community']['want'] > \ 

137 [v[0] for k, v in masters.items()][artist_index]: 

138 artists.pop(artist_index) 

139 masters.pop([master_id for master_id in masters][artist_index]) 

140 

141 # if current artist is not yet in the list, add the release 

142 if artist not in artists: 

143 masters[master.id] = [ 

144 master.data['community']['have'] + master.data['community']['want'], # album's popularity 

145 master.data['country'], # album's release country 

146 master.data['cover_image'], # album's cover image 

147 master.title.split(' - ')[1], # album's title 

148 artist, # album's artist 

149 master.year, # album's release year 

150 "/".join(master.data['style']), # album's list of genres 

151 "https://www.discogs.com" + master.data['uri'], # album's url on Discogs website 

152 ] 

153 masters = sorted(masters.items(), key=lambda x: x[1][0]) # sort all albums by popularity 

154 

155 top_10 = masters[len(masters) - 10:] # only the 10 most popular 

156 

157 # creates one database entry for each album in the top 10 

158 for release in top_10: 

159 choice = Choice( 

160 country=release[1][1], 

161 image=release[1][2], 

162 title=release[1][3], 

163 artist=release[1][4], 

164 year=release[1][5], 

165 genres=release[1][6], 

166 url=release[1][7], 

167 question_id=self.model.pk, 

168 ) 

169 choice.save() # saves to database 

170 

171 # renders template for details view of the newly created poll 

172 return HttpResponseRedirect(reverse('polls:detail', kwargs={'pk': self.model.pk}))

175@login_required # only logged-in users can access this function 

176def vote(request, question_id): # handles vote submission 

177 question = get_object_or_404(Question, pk=question_id) # if question cannot be found renders 404 page 

178 try: # tries to use choice_id from form 

179 selected_choice = question.choice_set.get(pk=request.POST['choice']) 

180 except (KeyError, Choice.DoesNotExist): # excepts if no choice_id was passed in 

181 return render(request, 'polls/detail.html', { # renders the question voting form again 

182 'question': question, # same question 

183 'error_message': "You didn't select a choice.", # error message to be added 

184 }) 

185 else: # once accepted 

186 selected_choice.votes += 1 # increases number of votes by one 

187 selected_choice.save() # saves to database 

188 # renders template for results view of the current poll 

189 return HttpResponseRedirect(reverse('polls:results', args=(question.id,)))
