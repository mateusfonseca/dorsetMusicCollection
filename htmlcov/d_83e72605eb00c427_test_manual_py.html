3""" 

4This file defines all the manual tests for the internal app polls. 

5Each test is a function that interacts with a certain view and evaluates its response 

6against a pre-defined assertion. If the assertion is correct, the test has passed. 

7If the assertion is incorrect, the test has failed. 

8Each test tests only one functionality of the view, for this reason, tests are grouped 

9together into classes. Each class represents a suite of tests for a particular view. 

10 

11Because the tests defined here are manual, user input is expected. This means that 

12execution stops and waits for the tester to input some data. The tags defined here 

13should be used to exclude these tests when automating. 

14""" 

15

41 def test_view_vote_method(self): # view should assign the vote to the correct object 

42 self.client.force_login(self.user) # logs-in in mock user ignoring credentials 

43 choices = Choice.objects.filter(question_id=self.question.id) # retrieves all choices associated with question 

44 

45 # CLI interaction block 

46 print("\nVOTE IN A POLL") 

47 

48 # displays list of choices and read input 

49 opt = "-1" 

50 while not opt.isnumeric() or int(opt) - 1 not in range(len(choices)): 

51 print(f"\n{str(self.question)}") 

52 print("Choices:") 

53 for i, c in enumerate(choices, start=1): 

54 print('{0}. {1} {2}'.format(i, str(c), f'votes: {c.votes}'))

58 # number of votes choice had before posting 

59 last_number_of_votes = choice.votes 

60

63 SERVER_NAME='localhost', secure=True) 

64 choice.refresh_from_db() # refreshes choice so that it reflects database after posting

66 self.assertEqual(response.status_code, 302) # expects 302 redirection after voting
